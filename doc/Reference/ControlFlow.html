def check_number(number)
  if number.positive?
    'positive'
  elsif number.negative?
    'negative'
  else
    'zero'
  end
end

def day_of_week(day)
  case day
  when 'Monday'
    'Start of the work week'
  when 'Saturday', 'Sunday'
    'Weekend'
  else
    'Midweek'
  end
end

    case 文の例: 曜日に応じたメッセージを返します。

    @param day [String] 曜日を表す文字列 (“Monday”, “Saturday”, など) @return [String] 曜日に応じたメッセージ

    使用例:

    control_flow = Reference::ControlFlow.new
puts control_flow.day_of_week("Monday") # => "Start of the work week"

def is_even(number) # rubocop:disable Naming/PredicateName
  number.even? ? 'Even' : 'Odd'
end
