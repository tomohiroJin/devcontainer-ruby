def each_loop(array)
  result = []
  array.each do |item|
    result << item * 2
  end
  result
end

def times_loop(times)

def until_loop(limit)
  count = 0
  result = []
  until count >= limit
    count += 1
    result << count
  end
  result
end

def while_loop(count)
  result = []
  while count > 0
    result << count
    count -= 1
  end
  result
end
